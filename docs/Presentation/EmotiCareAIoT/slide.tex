\section{Giới thiệu sản phẩm}

\begin{frame}{Định vị sản phẩm}
  \begin{block}{EmotiCare AIoT là gì?}
    Thiết bị AIoT giúp người dùng \textbf{nhận biết, thấu hiểu và quản lý cảm xúc}
    trong cuộc sống hằng ngày thông qua giọng nói.
  \end{block}
  \vspace{0.3cm}
  \begin{columns}[T]
    \column{0.5\textwidth}

      \item Nhận diện cảm xúc cốt lõi xử lý tại \textbf{Edge} (nhanh, riêng tư)
      \item Gợi ý, trò chuyện, báo cáo dài hạn dùng \textbf{Cloud}
      \item Mọi kết quả hiển thị trên \textbf{TFT screen}
    \end{itemize}
    \column{0.5\textwidth}
    \begin{center}

      \draw[-{Latex}] (edge) -- (cloud);
      \draw[-{Latex}] (cloud) -- (tft);
      \draw[-{Latex}] (edge) to[bend left=60] (tft);
    \end{tikzpicture}
    \end{center}
  \end{columns}
  \vspace{0.2cm}
  \centering \textit{``EmotiCare AIoT -- Understand your feelings, care for your mind.''}
\end{frame}

\begin{frame}{Mục đích sản phẩm}
  \begin{itemize}
    \item Nhận diện cảm xúc trong \textbf{15 giây} sau khi tương tác giọng nói hợp lệ
    \item Đưa ra hoạt động/bài hát/podcast/phản hồi đồng cảm trong \textbf{20 giây} (khi có Internet)
    \item Lưu lại từng phiên cảm xúc để xem lại lịch sử
    \item Tạo báo cáo cảm xúc theo \textbf{ngày, tuần, tháng, năm} trong \textbf{180 giây}
    \item Phân tích hiệu quả hoạt động/nội dung để gợi ý ngày càng phù hợp hơn
  \end{itemize}
\end{frame}

\begin{frame}{Tổng quan 3 SMART Objective}

  \hline

\begin{frame}{UC-01: Speech Emotion Recognition}
  \textbf{Input:} Giọng nói \quad \textbf{Output:} Trạng thái cảm xúc (vui vẻ, bình thường, căng thẳng, buồn bã, tức giận, mệt mỏi)
  \vspace{0.2cm}
  \begin{enumerate}
    \item Người dùng kích hoạt Check-in
    \item Edge AI tiền xử lý âm thanh, trích xuất đặc trưng
    \item Mô hình SER phân loại cảm xúc + confidence
    \item TFT hiển thị kết quả, lưu emotion session
  \end{enumerate}
  \begin{block}{Mục tiêu hiệu năng}
    Hoàn tất trong \textbf{15 giây}. Hoạt động được cả khi \textbf{offline}.
  \end{block}
\end{frame}

\begin{frame}{UC-02, UC-03, UC-04: Hỗ trợ qua Cloud (20s)}
  \small
  \begin{tabularx}{\textwidth}{|L{1.3cm}|L{3.2cm}|X|}
  \hline
  \textbf{UC} & \textbf{Tên} & \textbf{Mô tả ngắn} \\
  \hline
  UC-02 & Gợi ý hoạt động \& nội dung & Cloud đề xuất hoạt động, bài hát, podcast theo emotion label/lịch sử \\
  \hline
  UC-03 & Chọn bài hát/podcast theo chủ đích & Người dùng chọn category (thư giãn, tập trung, ngủ nghỉ,...) \\
  \hline
  UC-04 & Trò chuyện hỗ trợ cảm xúc & Phản hồi đồng cảm, có safety filter, xử lý tín hiệu nguy cấp \\
  \hline
  \end{tabularx}
  \vspace{0.2cm}
  \textit{Tất cả đều cần Internet; nếu mất kết nối, TFT báo trạng thái cần Cloud.}
\end{frame}

\begin{frame}{UC-05: Thống kê \& phân tích xu hướng cảm xúc}
  \textbf{Input:} Emotion sessions, activity logs, media logs, conversation metadata \\
  \textbf{Output:} Báo cáo rút gọn theo ngày/tuần/tháng/năm trên TFT

  \begin{itemize}

    \item Nén thành các \textbf{TFT report cards} ngắn gọn
    \item Nếu dữ liệu ít $\rightarrow$ trạng thái \texttt{limited\_data}
  \end{itemize}
  \begin{block}{Mục tiêu hiệu năng}
    Hiển thị trên TFT trong vòng \textbf{180 giây}.
  \end{block}
\end{frame}

    \item Đặc trưng chính: Log-Mel Spectrogram, MFCC, pitch, energy

  \hline
  neutral, calm & Bình thường & Trạng thái ổn định \\
  \hline
  happy & Vui vẻ & Cảm xúc tích cực \\
  \hline
  sad & Buồn bã & Cần phản hồi đồng cảm \\
  \hline
  angry & Tức giận & Cần gợi ý tạm dừng, thở chậm \\
  \hline
  fearful, surprised & Căng thẳng & Cần hỗ trợ giảm áp lực \\
  \hline
  -- & Mệt mỏi & Nhãn mở rộng, cần dữ liệu bổ sung \\
  \hline
  \end{tabularx}
\end{frame}

\begin{frame}{Logic confidence \& quality flag}
  \small
  \begin{tabularx}{\textwidth}{|X|X|}
  \hline
  \textbf{Điều kiện} & \textbf{Hành vi hệ thống} \\
  \hline
  confidence $\geq$ 0.75 \& clean & Hiển thị emotion label, chuyển sang hỗ trợ \\
  \hline
  0.50 $\leq$ confidence $<$ 0.75 & Hiển thị ``có thể là...'', cho xác nhận \\
  \hline
  confidence $<$ 0.50 & Không chắc chắn, không kết luận xu hướng \\
  \hline
  quality\_flag = too\_short & Yêu cầu ghi âm lại \\
  \hline
  quality\_flag = noisy & Cảnh báo môi trường nhiễu \\
  \hline
  \end{tabularx}
  \vspace{0.2cm}
  \textit{Đánh giá mô hình: Accuracy, Confusion matrix, Macro F1, Latency, Model size, Robustness test.}
\end{frame}
