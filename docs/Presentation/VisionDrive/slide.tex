\begin{frame}{Problem}
  \begin{block}{Motorbike riders need real-time attention support}
    In urban traffic, distraction can happen in seconds: drowsiness, looking away, phone usage, unstable handlebar motion.
  \end{block}
  \begin{itemize}
    \item Motorbikes are common in Vietnam, but driver monitoring systems mostly target cars.
    \item Smartphone safety apps can track routes, but phones themselves can cause distraction.
    \item Dashcams record incidents, but often do not prevent them in real time.
    \item Riders need a low-cost, edge-first system with immediate feedback.
  \end{itemize}
\end{frame}

\begin{frame}{Product Vision}
  \begin{columns}
    \column{0.58\textwidth}
    \textbf{VisionDriveAI} is an AIoT safety assistant for urban motorbike riders.
    \vspace{0.3cm}
    \begin{itemize}
      \item Detect drowsiness, gaze deviation, and phone holding.
      \item Combine camera and IMU for stronger context.

\begin{frame}{Objective 1: Real-Time Detection}
  \begin{columns}
    \column{0.48\textwidth}
    \begin{block}{Drowsiness + Gaze}
      \begin{itemize}
        \item Eye closure over threshold.
        \item Head nodding down.
        \item Looking away continuously.
      \end{itemize}
    \end{block}
    \column{0.48\textwidth}
    \begin{block}{Phone Holding}
      \begin{itemize}
        \item Detect hand-phone posture.
        \item Combine IMU to reduce false positives.
        \item Prioritize moving vehicle context.
      \end{itemize}
    \end{block}
  \end{columns}
  \vspace{0.3cm}
  \begin{center}
    \textbf{Target latency: under 500ms on edge device}
  \end{center}
\end{frame}

\begin{frame}{Objective 2: Alert and App Feedback}
  \begin{columns}
    \column{0.48\textwidth}
    \textbf{3-level escalation}
    \begin{enumerate}
      \item Light handlebar vibration.
      \item Strong vibration + short buzzer beep.
      \item Continuous buzzer + red LED.
    \end{enumerate}
    \column{0.48\textwidth}
    \textbf{App and AI pattern}
    \begin{itemize}
      \item Log event timestamp and type.
      \item Push notification for severe events.
      \item Detect risky time/route patterns.
      \item Adjust sensitivity over time.
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Objective 3: Trip Analytics}
  \begin{block}{Safety Score 0--100}
    After each trip, the app summarizes distraction events, eye-closure duration, phone holding, hard braking, and alert frequency.
  \end{block}
  \begin{itemize}
    \item Weekly/monthly trend dashboard.
    \item Clear behavior feedback for riders.
    \item Personalized baseline after 1--2 weeks of data.
    \item Lower false positives compared with fixed thresholds.
  \end{itemize}
\end{frame}

\begin{frame}{User Flow}
  \begin{center}
    \Large
    Mount Device $\rightarrow$ Pair App $\rightarrow$ Start Trip $\rightarrow$ Detect + Alert $\rightarrow$ Safety Score
  \end{center}
  \vspace{0.4cm}
  \begin{itemize}
    \item Edge Device can alert without Internet.
    \item App enables logs, push notifications, and trip dashboard.
    \item Server enables pattern analysis and personalization.
  \end{itemize}
\end{frame}

\begin{frame}{Business Value}
  \begin{itemize}
    \item \textbf{Safety:} detects risky behavior before it becomes an incident.
    \item \textbf{Affordability:} prototype under 400k VND component cost.
    \item \textbf{Local relevance:} designed for Vietnam's motorbike-heavy urban traffic.
    \item \textbf{Scalability:} useful for families, delivery riders, and small fleets.
    \item \textbf{Data advantage:} long-term patterns can guide safer habits.
  \end{itemize}
\end{frame}

\begin{frame}{Limitations and Roadmap}
  \begin{columns}
    \column{0.48\textwidth}
    \textbf{Limitations}
    \begin{itemize}
      \item Low light and rain affect camera.
      \item Edge AI has memory/compute limits.
      \item Privacy needs careful handling.
      \item Real-road calibration is required.
    \end{itemize}
    \column{0.48\textwidth}
    \textbf{Roadmap}
    \begin{itemize}
      \item Robust enclosure and mounting.
      \item Better sensor fusion.
      \item Face/hand model optimization.
      \item Map-based risk hotspot analytics.
      \item Personalized driver profiles.
    \end{itemize}
  \end{columns}
\end{frame}
